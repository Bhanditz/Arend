Tutorial¶

Here is the data type for booleans in Arend:

\import Data.Unit
\import Data.Empty

\data Bool | true | false

\func True (b : Bool) : \Prop
    | true => Unit
    | false => Empty

\func not (b : Bool) : Bool
    | true => false
    | false => true

\func if {A : \Type} (b : Bool) (then else : A) : A \elim b
    | true => then
    | false => else

Here is some fine algebra for you reading pleasure:

\import Data.Bool

\class Semigroup {
  | X : \Type0
  | op : X -> X -> X
  | assoc : \Pi (x y z : X) -> op (op x y) z = op x (op y z)
}

\func xor-semigroup => \new Semigroup { X => Bool | op => xor | assoc => {?} }

\func and-semigroup => \new Semigroup { X => Bool | op => and | assoc => {?} }

\class Monoid {
  | S : Semigroup
  | id : S.X
  | lunit : \Pi (x : S.X) -> S.op id x = x
  | runit : \Pi (x : S.X) -> S.op x id = x
}

\func xor-monoid => \new Monoid {
  | S => xor-semigroup
  | id => false
  | lunit => {?}
  | runit => {?}
}

\func and-monoid => \new Monoid {
  | S => and-semigroup
  | id => true
  | lunit => {?}
  | runit => {?}
}

\class Group {
  | M : Monoid
  | inv : M.S.X -> M.S.X
  | linv : \Pi (x : M.S.X) -> M.S.op (inv x) x = M.id
  | rinv : \Pi (x : M.S.X) -> M.S.op x (inv x) = M.id
}

\func xor-group => \new Group {
  | M => xor-monoid
  | inv => {?}
  | linv => {?}
  | rinv => {?}
}

\class AbelianGroup {
  | G : Group
  | comm : \Pi (x y : G.M.S.X) -> G.M.S.op x y = G.M.S.op y x
}

\func xor-abelian => \new AbelianGroup {
  | G => xor-group
  | comm => {?}
}
